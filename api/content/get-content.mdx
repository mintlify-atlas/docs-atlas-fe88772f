---
title: 'Get Content'
openapi: 'GET /api/content/{contentId}'
---

Retrieves detailed information about a specific published content item. This is a public endpoint that does not require authentication.

## Path Parameters

<ParamField path="contentId" type="string" required>
  The unique identifier of the content
</ParamField>

## Response

<ResponseField name="content" type="object">
  Content object with detailed information
  
  <Expandable title="Content Object">
    <ResponseField name="_id" type="string">
      Unique identifier for the content
    </ResponseField>
    
    <ResponseField name="title" type="string" required>
      Content title
    </ResponseField>
    
    <ResponseField name="description" type="string" required>
      Content description
    </ResponseField>
    
    <ResponseField name="type" type="string" required>
      Content type: `pdf`, `video`, or `audio`
    </ResponseField>
    
    <ResponseField name="category" type="string" required>
      Category ID
    </ResponseField>
    
    <ResponseField name="price" type="number" required>
      Content price (minimum: 0)
    </ResponseField>
    
    <ResponseField name="thumbnailUrl" type="string">
      URL to the content thumbnail image
    </ResponseField>
    
    <ResponseField name="duration" type="number">
      Duration in seconds (for video/audio content only)
    </ResponseField>
    
    <ResponseField name="fileSize" type="number">
      File size in bytes
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Content status: `draft` or `published`
    </ResponseField>
    
    <ResponseField name="tags" type="array">
      Array of tag strings
    </ResponseField>
    
    <ResponseField name="createdBy" type="object">
      Creator information
      
      <Expandable title="Creator Object">
        <ResponseField name="_id" type="string">
          User ID
        </ResponseField>
        
        <ResponseField name="name" type="string">
          Creator name
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="createdAt" type="string">
      ISO 8601 date string
    </ResponseField>
    
    <ResponseField name="updatedAt" type="string">
      ISO 8601 date string
    </ResponseField>
  </Expandable>
</ResponseField>

<Note>
  The `fileUrl` and `filePublicId` fields are not included in the response. To access the file URL, users must purchase the content and use the [Access Content](/api/content/access-content) endpoint.
</Note>

## Error Responses

<ResponseField name="error" type="string">
  Error message
</ResponseField>

### 404 Not Found
Returned when the content does not exist or is not published.

```json
{
  "error": "Content not found"
}
```
